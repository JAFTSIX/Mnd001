create table cliente(


id_cliente varchar(10) primary key,
nombre varchar(50),
apellido varchar(50)
);

create table cuenta(


id_cliente varchar(10) ,
id varchar(10) primary key,
username varchar(50),
pass varchar(300),
correo varchar(1000),
 constraint  fkc_id foreign key(id_CLIENTE) 
REFERENCES CLIENTE(id_CLIENTE)

);

create table tarjeta(

id_cliente varchar(10) ,
id varchar(10) primary key,
username varchar(50),
pass varchar(300),
correo varchar(1000),

CONSTRAINT FKT_CLIENTE FOREIGN KEY (id_CLIENTE) 
REFERENCES CLIENTE(id_CLIENTE)
);
